       public string FtpDownload()
        {
            try
            {
                FtpWebRequest readReq = (FtpWebRequest)WebRequest.Create(new Uri("ftp://112.95.245.253/NJCFDatas.txt"));
                readReq.Method = WebRequestMethods.Ftp.DownloadFile;
                readReq.Credentials = new NetworkCredential("movieposter", "201803");
                readReq.UseBinary = true;
                FtpWebResponse response = (FtpWebResponse)readReq.GetResponse();
                Stream stream = response.GetResponseStream();
                List<byte> data = new List<byte>();
                byte[] buffer = new byte[4096];
                int len = stream.Read(buffer, 0, buffer.Length);
                while (len > 0)
                {
                    data.AddRange(buffer.Take(len));
                    len = stream.Read(buffer, 0, buffer.Length);
                }
                stream.Close();
                response.Close();
                return Encoding.Default.GetString(data.ToArray());
            }
            catch (Exception ex)
            {
                AlarmCenter.DataCenter.DataCenter.WriteLogFile(ex.ToString());
                return null;
            }
        }
